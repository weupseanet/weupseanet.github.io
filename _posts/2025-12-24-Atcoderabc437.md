### B-Tombola
#### 题解
答案为 数列中寻找到I在x，将w[I]完成加一操作。
输出`ans = max(w[1] , w[2] , w[3] , w[4] , w[....]`
#### AC CODE
```cpp
#include <iostream>
#include <algorithm>
using namespace std;
int x , y , k;
int a[10010][10010] , w[10010];
int q , ans;
int main () {
	cin >> x >> y >> k;
	for(int i = 1;i <= x;i ++) 
		for(int j = 1;j <= y;j ++) 
			cin >> a[i][j];
	
	while(k --) {
		cin >> q;
		for(int i = 1;i <= x;i ++) 
			for(int j = 1;j <= y;j ++) 
				if(a[i][j] == q) w[i] ++;
	}
	
	for(int i = 1;i <= x;i ++) 
		ans = max(ans , w[i]);
	
	cout << ans;
	return 0;
}
```
### C - Reindeer and Sleigh 2
#### 题解
贪心思路，让力气小、体重小的鹿坐上车（先考虑体重），以此递减..
#### AC CODE
```cpp
#include <bits/stdc++.h>
using namespace std;

struct node {
    long long w, p;
} a[300005];

bool cmp(const node &x, const node &y) {
    return x.w + x.p < y.w + y.p;
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        int n;
        cin >> n;

        long long sum_p = 0, tmp = 0;
        for (int i = 1; i <= n; i++) {
            cin >> a[i].w >> a[i].p;
            sum_p += a[i].p;
        }

        sort(a + 1, a + n + 1, cmp);

        int ans = n;
        for (int i = 1; i <= n; i++) {
            tmp += a[i].w;
            sum_p -= a[i].p;
            if (tmp > sum_p) {
                ans = i - 1;
                break;
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
```
### D - Sum of Differences
#### 题解
裸裸诈骗题。
让你求
```c
for(ll i = 1;i <= n;i ++) 
		for(ll j = 1;j <= m;j ++) 
			ans = (ans + (abs(a[i] - b[j]) % MOD)) % MOD;
```
实际分段求：
```c
	for (ll i = 0; i < n; i++) {
		ll tmp_a = a[i];
		ll idx = upper_bound(b, b + m, tmp_a) - b;
		ll lsum = (tmp_a % MOD * idx % MOD - w[idx] + MOD) % MOD;
		ll rsum = (w[m] - w[idx] - tmp_a % MOD * (m - idx) % MOD + MOD) % MOD;
		
		ans = (ans + lsum + rsum) % MOD;
	}
```
（前缀和..）
#### AC CODE
```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll MOD = 998244353;
ll n, m, ans;
ll a[300005], b[300005], w[300005];

int main() {
	cin >> n >> m;
	for (ll i = 0; i < n; i++) cin >> a[i];
	for (ll i = 0; i < m; i++) cin >> b[i];
	
	sort(a, a + n);
	sort(b, b + m);
	
	for (ll i = 0; i < m; i++)
		w[i + 1] = (w[i] + b[i]) % MOD;
	
	for (ll i = 0; i < n; i++) {
		ll tmp_a = a[i];
		ll idx = upper_bound(b, b + m, tmp_a) - b;
		ll lsum = (tmp_a % MOD * idx % MOD - w[idx] + MOD) % MOD;
		ll rsum = (w[m] - w[idx] - tmp_a % MOD * (m - idx) % MOD + MOD) % MOD;
		
		ans = (ans + lsum + rsum) % MOD;
	}
	
	cout << ans % MOD;
	return 0;
}
```
